\begin{frame}{\dots\ but either misses the conflict or flattens it to $\mathcal{I}_c = 1$}
    \begin{exampleblock}{$\mathcal{I}_c$ under two encodings, neither informative}
        \centering
        \begin{tabular}{lcc}
            \toprule
            Planning failure                         & precondition & enriched \\
            \midrule
            Goal with no producer, as in Locked Door & $1$          & $1$      \\
            Two goals that never hold together       & $0$          & $1$      \\
            A goal that irreversibly blocks another  & $0$          & $1$      \\
            \bottomrule
        \end{tabular}
    \end{exampleblock}

    \begin{block}{Why neither helps}
        \begin{itemize}
            \item Without delete effects, operator-induced conflicts read as a false $0$
            \item With them, $B$ collapses each conflict to a flat $1$
            \item Chain depth, goal clashes, and reversibility distinctions vanish
        \end{itemize}
    \end{block}
