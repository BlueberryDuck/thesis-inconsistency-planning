\section{Three diagnostic profiles}

% BUDGET: ~90s
\begin{frame}{Every profile pairs a goal-level count with a structural count}
    \begin{exampleblock}{Diagnostic profile shape}
        \[
            \mathcal{I}(\Pi, h) = (\mathcal{I}^{\text{scope}}(\Pi, h),\ \mathcal{I}^{\text{struct}}(\Pi, h))
        \]
    \end{exampleblock}

    \begin{block}{Two components, one parameter}
        \begin{itemize}
            \item Scope counts goal propositions exhibiting the phenomenon
            \item Structure counts the underlying artifacts, such as unreachable propositions, mutex pairs, sequencing pairs
            \item Horizon $h \in \mathbb{N}$ bounds the reachability used to detect both counts
            \item Same shape instantiated by three measures $\mathcal{I}_{\text{UR}}$, $\mathcal{I}_{\text{MX}}$, $\mathcal{I}_{\text{GS}}$
        \end{itemize}
    \end{block}

    \sectioncite{Thesis \S4.4}
\end{frame}

% BUDGET: ~90s
\begin{frame}{$\mathcal{I}_{\text{UR}}$ counts blocked goals and the cascade behind them}
    \begin{exampleblock}{Locked Door at convergence}
        $S_{\text{reach}} = \{\{\text{at\_door}\}\}$, so 1 goal is blocked and 3 propositions are unreachable.
        \[
            \mathcal{I}_{\text{UR}}(\Pi) = (1, 3)
        \]
    \end{exampleblock}

    \begin{block}{Definition at horizon $h$}
        \begin{itemize}
            \item Scope counts goal propositions absent from every state in $S^h_{\text{reach}}$
            \item Structure counts all propositions absent from every state in $S^h_{\text{reach}}$
            \item Difference structure $-$ scope reveals cascading dependency depth
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:bank-vault}{Thesis \S4.4.3}
\end{frame}

% BUDGET: ~90s
\begin{frame}{$\mathcal{I}_{\text{MX}}$ separates isolated pairs from dense cliques}
    \begin{exampleblock}{Two scenarios, two profiles}
        \centering
        \begin{tabular}{lcc}
            \toprule
            Scenario      & $S_{\text{reach}}$                                        & $\mathcal{I}_{\text{MX}}(\Pi)$ \\
            \midrule
            Light Switch  & $\{\{\text{off}\}, \{\text{on}\}\}$                       & $(2, 1)$                       \\
            Traffic Light & $\{\{\text{red}\}, \{\text{green}\}, \{\text{yellow}\}\}$ & $(3, 3)$                       \\
            \bottomrule
        \end{tabular}
    \end{exampleblock}

    \begin{block}{Definition at horizon $h$}
        \begin{itemize}
            \item Goals $g_1, g_2$ are mutex when each is individually reachable in $S^h_{\text{reach}}$ but no state holds both
            \item Scope counts goal propositions in at least one mutex pair
            \item Structure counts unordered mutex pairs $\{g_1, g_2\} \subseteq G$
            \item A complete clique gives $\mathcal{I}^{\text{struct}}_{\text{MX}} = \binom{|G|}{2}$
        \end{itemize}
    \end{block}

    \sectioncite{Thesis \S4.4.4}
\end{frame}

% BUDGET: ~90s
\begin{frame}{$\mathcal{I}_{\text{GS}}$ measures the direction that mutex erases}
    \begin{exampleblock}{Trust and Travel at convergence}
        $S_{\text{reach}}$ contains $\{\text{at\_local}, \text{trust}, \text{partnership}\}$ and $\{\text{at\_local}, \text{opportunity}\}$, never both goals.
        \[
            \mathcal{I}_{\text{GS}}(\Pi) = (2, 1) \qquad \text{ordered pair } (\text{opportunity}, \text{partnership})
        \]
        Travel deletes \textit{trust} irreversibly, so opportunity-states cannot reach partnership-states. The reverse direction is not blocked.
    \end{exampleblock}

    \begin{block}{Definition at horizon $h$}
        \begin{itemize}
            \item $(g_1, g_2)$ is a sequencing conflict when both are individually reachable yet no trace $\sigma_0, \ldots, \sigma_k$ achieves $g_1$ then $g_2$ within $k \leq h$
            \item Scope counts goals appearing in any ordered conflict pair
            \item Structure counts ordered pairs, so direction matters
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:trust-travel}{Thesis \S4.4.5}
\end{frame}

% BUDGET: ~60s
\begin{frame}{Sequencing implies mutex, unreachability sits aside}
    \begin{exampleblock}{Containment at horizon $h$}
        \[
            \{ (g_1, g_2) \text{ in } \mathcal{I}_{\text{GS}} \} \;\subsetneq\; \{ \{g_1, g_2\} \text{ in } \mathcal{I}_{\text{MX}} \}
        \]
        Unreachable goals participate in neither.
    \end{exampleblock}

    \begin{block}{What containment means for the diagnostic}
        \begin{itemize}
            \item Every sequencing conflict is also a mutex pair
            \item Light Switch is mutex without sequencing, reversible toggles
            \item Trust and Travel is mutex with sequencing, irreversible travel
            \item An unreachable goal blocks the chain before any pair test fires
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:seq-implies-mx}{Thesis \S4.5.2}
\end{frame}
