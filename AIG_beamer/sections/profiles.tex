\section{Diagnostic profiles and measures}

% BUDGET: ~75s
\begin{frame}{Reachable states and horizon-bounded reachability}
    All six measures are computed over horizon-bounded reachability, the horizon $h$ avoiding the worst-case exponential blowup of full reachability.

    \begin{block}{Reachable states}
        $S^0_{\text{reach}} = \{I\}$ \\[2pt]
        $S^{i+1}_{\text{reach}} = S^i_{\text{reach}} \cup \{\text{RESULT}(s, o) \mid s \in S^i_{\text{reach}},\ o \in O,\ \text{precond}(o) \subseteq s\}$
    \end{block}

    \begin{block}{Horizon-bounded reachability}
        \small
        \begin{itemize}
            \item Ascending chain $S^0_{\text{reach}} \subseteq S^1_{\text{reach}} \subseteq \cdots$ converges at $h^* \leq 2^{|P|}$, each step only adds states
            \item At $h \geq h^*$ the bound is exact, so detected conflicts are genuine, not short-horizon artifacts
            \item $h^*$ is worst-case exponential, so the evaluation bounds it at $h = 20$
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:horizon}{Thesis \S4.4.2}
\end{frame}

\begin{frame}{Diagnostic profiles and measures}
    \begin{exampleblock}{Diagnostic profile}
        \[
            \mathcal{I}(\Pi, h) = (\mathcal{I}^{\text{scope}}(\Pi, h),\ \mathcal{I}^{\text{struct}}(\Pi, h))
        \]
    \end{exampleblock}

    \begin{block}{Diagnostic measures}
        \begin{itemize}
            \item Scope measures count the affected goal propositions, structure measures count the underlying structural relationships
            \item Horizon $h \in \mathbb{N}$ bounds the reachability behind both counts
            \item Three profiles share this shape, the Unreachability Profile $\mathcal{I}_{\text{UR}}$, the Goal Mutex Profile $\mathcal{I}_{\text{MX}}$, and the Goal Sequencing Conflict Profile $\mathcal{I}_{\text{GS}}$
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:normalization}{Thesis \S4.4, \S4.6}
\end{frame}

\begin{frame}{$\mathcal{I}_{\text{UR}}$ counts unreachable propositions}
    A goal proposition is unreachable when no state within horizon $h$ contains it.

    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{block}{Definitions}
                \footnotesize
                \begin{align*}
                    \mathcal{I}^{\text{scope}}_{\text{UR}}(\Pi, h)  & = |\{g \in G \mid \nexists s \in S^h_{\text{reach}} : g \in s\}| \\
                    \mathcal{I}^{\text{struct}}_{\text{UR}}(\Pi, h) & = |\{p \in P \mid \nexists s \in S^h_{\text{reach}} : p \in s\}|
                \end{align*}
            \end{block}
        \end{column}

\begin{frame}{$\mathcal{I}_{\text{MX}}$ counts mutually exclusive goals}
    Two goals are mutex at horizon $h$, written $\text{mutex}_h(g_1, g_2)$, when each is reachable on its own but no reachable state holds both.

    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{block}{Definitions}
                \footnotesize
                \begin{align*}
                    \mathcal{I}^{\text{scope}}_{\text{MX}}(\Pi, h)  & = |\{g \in G \mid \exists g' \in G,\ g \neq g' :      \\
                                                                    & \quad \text{mutex}_h(g, g')\}|                        \\
                    \mathcal{I}^{\text{struct}}_{\text{MX}}(\Pi, h) & = |\{\{g_1, g_2\} \mid g_1, g_2 \in G,\ g_1 \neq g_2, \\
                                                                    & \quad \text{mutex}_h(g_1, g_2)\}|
                \end{align*}
            \end{block}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering
            \input{figures/fig_rover_mutex}\\[2pt]
            {\small $\mathcal{I}_{\text{MX}} = (4, 2)$}
        \end{column}
    \end{columns}

\end{frame}

\begin{frame}{$\mathcal{I}_{\text{GS}}$ counts sequencing conflicts}
    An ordered pair $(g_1, g_2)$ is a sequencing conflict at horizon $h$, written $\text{sc}_h(g_1, g_2)$, when both goals are reachable yet no trace $\sigma_0, \ldots, \sigma_k$ with $k \leq h$ achieves $g_1$ and then $g_2$.

    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{block}{Definitions}
                \footnotesize
                \begin{align*}
                    \mathcal{I}^{\text{scope}}_{\text{GS}}(\Pi, h)  & = |\{g \in G \mid \exists g' \in G :                       \\
                                                                    & \quad \text{sc}_h(g, g') \text{ or } \text{sc}_h(g', g)\}| \\
                    \mathcal{I}^{\text{struct}}_{\text{GS}}(\Pi, h) & = |\{(g_1, g_2) \mid g_1, g_2 \in G,\ g_1 \neq g_2,        \\
                                                                    & \quad \text{sc}_h(g_1, g_2)\}|
                \end{align*}
            \end{block}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering
