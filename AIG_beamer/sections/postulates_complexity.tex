\begin{frame}{PSPACE absorbs the multi-tier complexity landscape}
    \begin{block}{Decision-problem complexity for each $\mathcal{I} \in \{\mathcal{I}_{\text{UR}}, \mathcal{I}_{\text{MX}}, \mathcal{I}_{\text{GS}}\}$}
        \centering
        \footnotesize
        \begin{tabular}{lll}
            \toprule
            Problem & Planning        & Propositional $\mathcal{I}_c$ [Thimm and Wallner, 2016] \\
            \midrule
            UPPER   & PSPACE-complete & NP-complete                                             \\
            LOWER   & PSPACE-complete & coNP-complete                                           \\
            EXACT   & PSPACE-complete & $D^p$-complete                                          \\
            VALUE   & FPSPACE         & FP$^{\text{NP}[\log n]}$-complete                       \\
            \bottomrule
        \end{tabular}
    \end{block}

    \begin{block}{Why the landscape flattens}
        \begin{itemize}
            \item Propositional landscape spans three tiers, $\mathcal{I}_c$ shows the lowest
            \item Reachability over the state space is already PSPACE-complete [Bylander, 1994]
            \item PSPACE $=$ co-PSPACE collapses UPPER, LOWER, and EXACT into one class
            \item VALUE follows by polynomially many PSPACE calls
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:pspace-proof}{Thesis \S4.5.3}
\end{frame}
