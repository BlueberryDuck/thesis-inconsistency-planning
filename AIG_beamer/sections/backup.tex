\begin{frame}[noframenumbering]{Backup, Trust and Travel full reachability}
    \hypertarget{bk:trust-travel}{}
    \begin{exampleblock}{Six reachable states}
        \small
        $S_{\text{reach}} = \{ \{\text{at\_local}, \text{trust}\},\ \{\text{at\_local}, \text{trust}, \text{partnership}\},\ \{\text{at\_local}\},$\\
        $\phantom{S_{\text{reach}} = \{}\{\text{at\_local}, \text{opportunity}\},\ \{\text{at\_remote}\},\ \{\text{at\_remote}, \text{opportunity}\}\}$
    \end{exampleblock}

    \begin{block}{Why the conflict is one-directional}
        \begin{itemize}
            \item \textit{travel} deletes \textit{at\_local}, \textit{trust}, and \textit{partnership} on every firing
            \item From an \textit{opportunity}-state \textit{trust} is gone, so \textit{partnership} is unreachable
            \item From a \textit{partnership}-state \textit{travel} $\rightarrow$ \textit{exploit} still reaches \textit{opportunity}
            \item Conflict $(\text{opportunity}, \text{partnership})$ holds, reverse direction fails
        \end{itemize}
    \end{block}

    \sectioncite{Thesis \S4.3.3 Scenario~5}
\end{frame}

\begin{frame}[noframenumbering]{Backup, 3unsat and the $k$-tuple blind spot}
    \hypertarget{bk:3unsat}{}
    \begin{exampleblock}{Pairwise coexistence with no triple-satisfying state}
        \[
            G = \{g_1, g_2, g_3\}, \qquad
            S_{\text{reach}} \supseteq \{\{g_1, g_2\}, \{g_2, g_3\}, \{g_1, g_3\}\}
        \]
        Every pair is reachable together, yet no state satisfies all three.
    \end{exampleblock}

    \begin{block}{Consequence for the measures}
        \begin{itemize}
            \item All goals individually reachable, so $\mathcal{I}_{\text{UR}}(\Pi) = (0, 0)$
            \item No mutex pair exists in $S_{\text{reach}}$, so $\mathcal{I}_{\text{MX}}(\Pi) = (0, 0)$
            \item No ordered sequencing failure either, so $\mathcal{I}_{\text{GS}}(\Pi) = (0, 0)$
            \item All 20 completed \textit{3unsat} instances confirm this at every tested horizon
            \item Future work extends to $k$-tuple analysis for $|G| \geq 3$ failures
        \end{itemize}
    \end{block}

    \sectioncite{Thesis \S6.3.6, \S7.4}
\end{frame}

\begin{frame}[noframenumbering]{Backup, sequencing implies mutex}
    \hypertarget{bk:seq-implies-mx}{}
    \begin{exampleblock}{Proposition}
        For any $\Pi$ and horizon $h$, if $(g_1, g_2)$ is a sequencing conflict with $g_2$ achievable, then $\{g_1, g_2\}$ is a mutex pair at horizon $h$.
    \end{exampleblock}

    \begin{block}{Proof sketch and counterexample}
        \begin{itemize}
            \item Sequencing gives both individually achievable, and no trace has $g_1$ at $\sigma_i$ and $g_2$ at $\sigma_j$ with $i \leq j$
            \item The $i = j$ case is the mutex condition, no reachable state holds both
            \item Combined with achievability, this is the mutex definition
            \item Converse fails on Light Switch, $\mathcal{I}_{\text{MX}} = (2, 1)$ yet $\mathcal{I}_{\text{GS}} = (0, 0)$ because toggles are reversible
        \end{itemize}
    \end{block}

    \sectioncite{Thesis \S4.5.2}
\end{frame}

\begin{frame}[noframenumbering,fragile]{Backup, witness rules in ASP}
    \hypertarget{bk:asp-listing}{}
    \scriptsize
    \begin{verbatim}
% True reachability of a proposition
true_reachable(P) :- holds(P, T), prop(P).

% Coexistence witness for a goal pair
coexist_witness(G1, G2) :- goal(G1), goal(G2), G1 < G2,
    holds(G1, T), holds(G2, T), time(T).

% Sequencing witness, G2 after G1
g2_after_g1_witness(G1, G2) :- goal(G1), goal(G2), G1 != G2,
    holds(G1, T1), holds(G2, T2), T2 >= T1.
\end{verbatim}
    \normalsize
    Brave reasoning unions these atoms across answer sets. An atom missing under brave means no trace satisfies the condition.

    \sectioncite{Thesis \S5.3.3}
\end{frame}

\begin{frame}[noframenumbering]{Backup, PSPACE-completeness proof sketch}
    \hypertarget{bk:pspace-proof}{}
    \begin{block}{Membership of $\text{UPPER}_{\mathcal{I}_{\text{UR}}}$ in PSPACE}
        \begin{itemize}
            \item Nondeterministically guess an operator trace from $I$ in polynomial space, accept when $g$ appears
            \item Acyclic paths have length $\leq 2^{|P|}$, so a polynomial-bit step counter suffices
            \item NPSPACE $=$ PSPACE by Savitch, iterating over $|G|$ goals stays polynomial
        \end{itemize}
    \end{block}

    \begin{block}{Hardness via reduction from PlanSat}
        From $\Pi = \langle P, O, I, G \rangle$, build $\Pi' = \langle P \cup \{g^*\},\ O \cup \{o^*\},\ I,\ \{g^*\} \rangle$ with $o^* = \langle G, \{g^*\}, \emptyset \rangle$. Then $g^*$ is reachable iff $\Pi$ has a plan, so $\text{UPPER}_{\mathcal{I}_{\text{UR}}}$ at $k = 0$ decides PlanSat.
    \end{block}

    \sectioncite{Thesis \S4.5.3}
\end{frame}

\begin{frame}[noframenumbering]{Backup, why ASP over SAT and MaxSAT}
    \hypertarget{bk:asp-vs-sat}{}
    \begin{block}{Reasons in favor of ASP}
        \begin{itemize}
            \item Default negation supports non-monotonic reasoning, matching reversed Operator Monotonicity
            \item Brave reasoning unions answer-set traces in one solver call, no manual enumeration
            \item ASP outperforms basic SAT encodings for inconsistency measures [Kuhlmann et al., 2025]
        \end{itemize}
    \end{block}

    \begin{block}{Trade-off acknowledged}
        Specialized MaxSAT beats ASP on classical formula-counting measures [Niskanen et al., 2023], and remains a future-work direction for scalability on planning-native measures.
    \end{block}

    \sectioncite{Thesis \S5, \S7.4}
\end{frame}

        \end{itemize}

        \begin{itemize}
            \item Solving dominates in 6 of 7 domains, grounding dominates only on \textit{pegsol-row5}
            \item \textit{document-transfer} prob06 spends 0.12\,s grounding and 18.8\,s solving, ratio 154
            \item Solve cost reflects combinatorial brave search across delete effects, $O(|O| \times h)$ grounding bound stays separate
