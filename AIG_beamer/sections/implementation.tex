\begin{frame}{Witness absence under brave reasoning is a universal proof}
    \begin{block}{Two witness atoms for goal pairs}
        \begin{itemize}
            \item \texttt{coexist\_witness($g_1$, $g_2$)} derived when some trace reaches a state with both goals
            \item \texttt{g2\_after\_g1\_witness($g_1$, $g_2$)} derived when some trace reaches $g_1$ then $g_2$
            \item Rover, neither witness fires for \textit{mapped} and \textit{in\_orbit}, so they are mutex and \textit{in\_orbit} blocks \textit{mapped}
        \end{itemize}
    \end{block}

    \begin{block}{Why brave is the right mode}
        \begin{itemize}
            \item A witness is in the brave union iff \emph{some} trace satisfies the condition, and absent iff \emph{no} trace does, exactly the mutex/sequencing definition
            \item Cautious would require \emph{every} trace to carry the witness, the wrong quantifier for a non-existence claim
        \end{itemize}
    \end{block}

\end{frame}
