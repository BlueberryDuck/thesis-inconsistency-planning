\begin{frame}{Inconsistency measure $\mathcal{I}(K)$}
    \begin{block}{Inconsistency measure $\mathcal{I} : \mathbb{K} \to \mathbb{R}_{\geq 0}$ [Thimm, 2019]}
        \begin{itemize}
            \item Higher value means more inconsistency, $\mathcal{I}(K) = 0$ iff $K$ consistent
            \item Rationality postulates such as Monotonicity and Free Formula Independence pin down sensible measures
            \item Contension $\mathcal{I}_c$ [Grant and Hunter, 2011] counts atoms forced to ``both true and false'', not formulas
        \end{itemize}
    \end{block}

    \begin{exampleblock}{Contension $\mathcal{I}_c$}
        $K = \{a \land b,\ \neg a,\ \neg b\}$ forces $a$ and $b$ to ``both'', so $\mathcal{I}_c(K) = 2$.
    \end{exampleblock}

\end{frame}

\begin{frame}{Classical planning problem $\Pi = \langle P, O, I, G \rangle$}
    \begin{block}{\acs{STRIPS} notation $\Pi = \langle P, O, I, G \rangle$ [Bylander 1994]}
        \footnotesize
        \begin{itemize}
            \item $P$ propositions, $O$ operators, $I \subseteq P$ initial state, $G \subseteq P$ goal
            \item Operator $o = \langle \text{precond}(o), \text{add}(o), \text{delete}(o) \rangle$, applicable when $\text{precond}(o) \subseteq s$, with result $\text{RESULT}(s, o) = (s \setminus \text{delete}(o)) \cup \text{add}(o)$
            \item Plan: applicable sequence from $I$ reaching a state $\supseteq G$
        \end{itemize}
    \end{block}
    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{exampleblock}{Rover Mission, the drill subproblem}
                \footnotesize
                $P = \{\text{arm\_deployed},\ \text{drill\_bit},\ \text{hole\_drilled},\ \text{soil\_sampled}\}$ \\[2pt]
                $O = \{\ \text{drill} = \langle \{\text{arm\_deployed}, \text{drill\_bit}\},\ \{\text{hole\_drilled}\},\ \emptyset \rangle,$ \\[1pt]
                $\hphantom{O = \{\ } \text{sample} = \langle \{\text{hole\_drilled}\},\ \{\text{soil\_sampled}\},\ \emptyset \rangle\ \}$ \\[2pt]
                $I = \{\text{arm\_deployed}\}, \qquad G = \{\text{soil\_sampled}\}$
            \end{exampleblock}

\begin{frame}{Answer Set Programming (ASP)}
    The measures test reachability over the state space, the kind of declarative search \ac{ASP} is built for.

    \begin{block}{Logic program [Dimopoulos et al., 1997]}
        \begin{itemize}
            \item Rule form $a \leftarrow b_1, \ldots, b_m, \mathbf{not}\ c_1, \ldots, \mathbf{not}\ c_n$
            \item An answer set is a minimal stable model under default negation
            \item Brave reasoning unions all answer sets, cautious intersects them [Eiter and Gottlob, 1995]
        \end{itemize}
    \end{block}

    \begin{exampleblock}{Two answer sets, one program}
        Rules \quad $p \leftarrow \mathbf{not}\ q.$ \quad $q \leftarrow \mathbf{not}\ p.$\\[2pt]
        Two answer sets $\{p\}$ and $\{q\}$. \quad
        $\text{BRAVE} = \{p, q\}$, \quad $\text{CAUTIOUS} = \emptyset$.
    \end{exampleblock}

\end{frame}
