\begin{frame}{From binary inconsistency to a numerical score $\mathcal{I}(K)$}
    \begin{exampleblock}{Knowledge base value}
        $K = \{a \land b,\ \neg a,\ \neg b\}$ \quad inconsistent.\\
        Contension measure [Grant and Hunter, 2011] yields $\mathcal{I}_c(K) = 2$.
    \end{exampleblock}

    \begin{block}{Inconsistency measure [Thimm, 2019]}
        \[
            \mathcal{I} : \mathbb{K} \to \mathbb{R}_{\geq 0}
        \]
        \begin{itemize}
            \item $\mathcal{I}(K) = 0$ iff $K$ consistent, the \emph{Consistency} postulate
            \item Monotone in formula additions and insensitive to free formulas
            \item Many candidates exist. This thesis tries $\mathcal{I}_c$ first on planning
        \end{itemize}
    \end{block}
    \sectioncite{Thesis \S2.1}
\end{frame}

% BUDGET: ~75s
\begin{frame}{Horizon $h$ bounds reachability and converges at $h^*$}
    \begin{exampleblock}{Horizon-bounded reachable states}
        \[
            S^h_{\text{reach}} = \{s \mid s \text{ reachable from } I \text{ in at most } h \text{ operator applications}\}
        \]
    \end{exampleblock}

    \begin{block}{Monotone chain, finite convergence}
        \begin{itemize}
            \item $S^0_{\text{reach}} \subseteq S^1_{\text{reach}} \subseteq \cdots$ non-decreasing in $h$
            \item Convergence horizon $h^* = \min\{h : S^h_{\text{reach}} = S_{\text{reach}}\}$, bounded by $h^* \leq 2^{|P|}$
            \item Planning Consistency requires $h \geq h^*$; longer $h$ adds grounding cost
            \item Evaluation fixes $h = 20$ on empirical grounds
        \end{itemize}
    \end{block}

    \sectioncitebk{bk:horizon}{Thesis \S4.4.2}
\end{frame}

% BUDGET: ~90s
\begin{frame}{Each answer set is a trace and brave reasoning unions them}
    \begin{exampleblock}{Two answer sets, one program}
        Rules \quad $p \leftarrow \mathbf{not}\ q.$ \quad $q \leftarrow \mathbf{not}\ p.$\\[2pt]
        Two answer sets $\{p\}$ and $\{q\}$. \quad
        $\text{BRAVE} = \{p, q\}$, \quad $\text{CAUTIOUS} = \emptyset$.
    \end{exampleblock}

    \begin{block}{Logic program [Dimopoulos et al., 1997]}
        \begin{itemize}
            \item Rule form $h \leftarrow b_1, \ldots, b_m, \mathbf{not}\ c_1, \ldots, \mathbf{not}\ c_n$
            \item An answer set is a minimal stable model under default negation
            \item Brave reasoning unions all answer sets, cautious intersects them [Eiter and Gottlob, 1995]
        \end{itemize}
    \end{block}

    \alert{Later, each answer set witnesses one execution trace and brave decides achievability.}

    \sectioncite{Thesis \S2.4}
\end{frame}
