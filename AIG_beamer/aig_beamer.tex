\section{Problem Motivation}

\begin{frame}{Planning in Practice: When Things Go Wrong}
    \begin{exampleblock}{Real-World Scenario: Logistics Planning}
        A delivery company needs to route packages through distribution centers:
        \begin{itemize}
            \item Package must be delivered to New York by noon
            \item Same package must be processed in Los Angeles at the same time
            \item System reports: \yellowhighlight{``Unsolvable''}
        \end{itemize}
        \textbf{Question}: Which constraint is causing the conflict?
    \end{exampleblock}

    \begin{alertblock}{The Diagnostic Gap}
        \begin{itemize}
            \item Planning systems excel when solutions exist
            \item But provide minimal diagnostic information when problems are unsolvable % Eriksson2018, Sreedharan2019
            \item No quantitative analysis of \textit{why} or \textit{where} conflicts occur % Goebelbecker2010, Sreedharan2019
        \end{itemize}
    \end{alertblock}
\end{frame}

\section{Technical Foundations}

\begin{frame}{Classical Planning}
    \begin{block}{Planning Problem} % Russell2016, Bylander1994
        A planning problem consists of:
        \begin{itemize}
            \item Facts: What can be true about the world % e.g., at(robot, room1), holding(block)
            \item Actions: Available operations % e.g., move, pickup, putdown
            \item Initial state: What's true at the start
            \item Goal: What must be achieved
        \end{itemize}
    \end{block}

    \begin{block}{Key Challenge}
        Determining solvability is PSPACE-complete % Bylander1994
        \begin{itemize}
            \item Exponential time and space complexity in worst case
            \item Makes debugging unsolvable problems computationally hard
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Taxonomy of Planning Inconsistencies}
    \begin{block}{Where Can Planning Go Wrong?} % Kuhlmann2025
        \begin{itemize}
            \item Domain-level: Domain specification itself is inconsistent
            \item Task-level: Planning problem is unsolvable (no valid plan exists)
            \item Plan-level inconsistencies:
                  \begin{itemize}
                      \item Execution-level: Plan contains non-applicable actions
                      \item Goal-level: Plan is executable but fails to achieve goals
                  \end{itemize}
        \end{itemize}
    \end{block}

    \begin{block}{Thesis Focus}
        Task-level inconsistency: Quantifying the degree of inconsistency in planning problems where no solution exists
    \end{block}
\end{frame}

\begin{frame}{Inconsistency Measurement}
    \begin{block}{From Propositional Logic} % Thimm2019 survey
        Rather than treating inconsistency as binary, assign numerical values that reflect severity of conflicts:
        \begin{itemize}
            \item Minimal inconsistent and maximal consistent set-based measures % "Shapely" Hunter2008
            \item Distance-based measures % Grant2013, Grant2017
            \item Hitting-set measure % Thimm2016
            \item Contension-based measure % Grant2011, Thimm2016
        \end{itemize}
    \end{block}

    \begin{exampleblock}{Adaptations for Other Domains}
        \begin{itemize}
            \item Database repairs
            \item Non-monotonic reasoning
            \item Process specifications using temporal logic % Corea2022
                  % Not yet systematically adapted to planning
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{Answer Set Programming}
    \begin{block}{\ac{ASP} for Inconsistency Measurement in Planning}
        \begin{itemize}
            \item Existing \ac{ASP} inconsistency measures provide adaptation foundation % Ulbricht2016
            \item \ac{ASP} algorithms outperform \ac{SAT} for computing inconsistency measures % Kuhlmann2025
            \item Non-monotonic semantics essential for planning's frame problem % Ulbricht2016
                  % frame problem: determining what remains unchanged after actions
            \item \ac{ASP} already used for planning (plasp 3 translates \ac{PDDL} to \ac{ASP}) % Dimopoulos2019
        \end{itemize}
    \end{block}

    \begin{block}{Thesis Approach}
        Adapt existing \ac{ASP} inconsistency measures to planning problems
    \end{block}
\end{frame}

\section{Research Approach}

\begin{frame}{Central Research Question}
    \Large How can inconsistency measures from propositional logic be meaningfully adapted to classical planning to provide diagnostic value for unsolvable planning problems?
\end{frame}

\begin{frame}{Research Questions}
    \begin{block}{RQ1: Which measures can be adapted and what postulates do they retain?}
        \begin{itemize}
            \item Survey existing inconsistency measures
            \item Analyze differences between propositional logic and planning
            \item Formal adaptation of 2-3 selected measures to \ac{STRIPS}
            \item Analyze postulate preservation in classical planning
            \item Formal definitions and examples
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Research Questions}
    \begin{block}{RQ2: How can adapted measures be efficiently computed using \ac{ASP}?}
        % Ensures the adapted measures don't exceed what ASP solvers can handle.
        % up to the second level of the polynomial hierarchy
        % Encoding vs Prototype: Encoding = ASP rules/algorithm; Prototype = full system (parser + translation + solver)
        \begin{itemize}
            \item Complexity analysis for \ac{ASP} compatibility
            \item Design \ac{ASP}-based encoding for at least one measure
            \item Develop parsing algorithms for \ac{STRIPS} and optionally \ac{PDDL}
            \item Prototype implementation for at least one measure
        \end{itemize}
    \end{block}

    \begin{block}{RQ3: Do adapted measures provide practical diagnostic value?}
        \begin{itemize}
            \item Experimental framework using benchmark planning problems
            \item Computational feasibility and scalability evaluation
            \item Diagnostic utility analysis for problem types
            \item Optionally assess impact on problem repair strategies
        \end{itemize}
    \end{block}
\end{frame}
